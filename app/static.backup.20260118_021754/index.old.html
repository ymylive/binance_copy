Cookie 登录

在 Binance 网页控制台执行命令导出 cookies.json，再上传启用。

一键导出 Cookie（在 Binance 控制台执行）
(async()=>{let c=[];const warn="注意：此命令无法导出 HttpOnly 登录态 Cookie（p20t/r20t/d1og 等）。如需完整登录态，请用 CDP/Playwright 导出。";if(window.cookieStore&&window.cookieStore.getAll){try{const r=await window.cookieStore.getAll();c=r.map(k=>({name:k.name,value:k.value,domain:k.domain||location.hostname,path:k.path||"/"}));}catch(e){}}if(!c.length){c=document.cookie.split(";").map(s=>s.trim()).filter(Boolean).map(s=>{const i=s.indexOf("=");return{name:s.slice(0,i),value:s.slice(i+1),domain:location.hostname,path:"/"}});}const j=JSON.stringify(c,null,2);const b=new Blob([j],{type:"application/json"});const a=document.createElement("a");a.href=URL.createObjectURL(b);a.download="cookies.json";a.click();URL.revokeObjectURL(a.href);if(navigator.clipboard&&navigator.clipboard.writeText){try{await navigator.clipboard.writeText(j);}catch(e){}}console.warn(warn);console.log("exported cookies:",c.length);})();
此命令只能导出非 HttpOnly Cookie，登录态可能不完整。如需完整登录态，请用 CDP/Playwright 导出 cookies_full.json 或使用 EditThisCookie 导出。
